nvm install in MAC

  1. open Terminal
  2. curl -o- https://raw.githubusercontent.com/creationix/nvm/v0.33.2/install.sh | bash
  3. open -e .bash_profile
  4. add export NVM_DIR="$HOME/.nvm" [ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh" in profile file
